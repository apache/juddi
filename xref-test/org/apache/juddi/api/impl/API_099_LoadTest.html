52          private static Log logger = LogFactory.getLog(API_080_SubscriptionTest.class);

54          private static API_010_PublisherTest api010 = new API_010_PublisherTest();
55          private static TckTModel tckTModel = new TckTModel(new UDDIPublicationImpl(), new UDDIInquiryImpl());
56          private static TckBusiness tckBusiness = new TckBusiness(new UDDIPublicationImpl(), new UDDIInquiryImpl());
57          private static TckBusinessService tckBusinessService = new TckBusinessService(new UDDIPublicationImpl(), new UDDIInquiryImpl());

59          UDDISubscriptionImpl sub = new UDDISubscriptionImpl();
60  
61          private static String authInfoJoe = null;
62          private static String authInfoSam = null;
63  
64          @BeforeClass
65          public static void setup() throws ConfigurationException {
66                  Registry.start();
67                  logger.info("API_080_SubscriptionTest");
68                  logger.debug("Getting auth token..");
69                  try {
70                          api010.saveJoePublisher();
71                          authInfoJoe = TckSecurity.getAuthToken(new UDDISecurityImpl(), TckPublisher.getJoePublisherId(), TckPublisher.getJoePassword());
72                  } catch (RemoteException e) {
73                          logger.error(e.getMessage(), e);
74                          Assert.fail("Could not obtain authInfo token.");
75                  }
76  
77          }
78  
79          @AfterClass
80          public static void stopRegistry() throws ConfigurationException {
81                  tckTModel.deleteCreatedTModels(authInfoJoe);
82                  Registry.stop();
83          }
84  
85          /**
86           * create 1000 subscriptions
87           * @throws Exception 
88           */
89          //@Ignore
90          @Test
91          public void joePublisher() throws Exception {
92                  //Assume.assumeTrue(TckPublisher.isLoadTest());
93                  List<String> keys = new ArrayList<String>();
94                  for (int i = 0; i < 1000; i++) {
95                          Holder<List<Subscription>> items = new Holder<List<Subscription>>();
96                          items.value = new ArrayList<Subscription>();
97                          Subscription s = new Subscription();
98                          s.setSubscriptionFilter(new SubscriptionFilter());
99                          s.getSubscriptionFilter().setFindBusiness(new FindBusiness());
100                         s.getSubscriptionFilter().getFindBusiness().setFindQualifiers(new FindQualifiers());
101                         s.getSubscriptionFilter().getFindBusiness().getFindQualifiers().getFindQualifier().add(UDDIConstants.APPROXIMATE_MATCH);
102                         s.getSubscriptionFilter().getFindBusiness().getName().add(new Name(UDDIConstants.WILDCARD, null));
103                         items.value.add(s);
104                         sub.saveSubscription(authInfoJoe, items);
105                         keys.add(items.value.get(0).getSubscriptionKey());
106                 }
107                 List<Subscription> subscriptions = sub.getSubscriptions(authInfoJoe);
108                 DeleteSubscription ds = new DeleteSubscription();
109                 ds.setAuthInfo(authInfoJoe);
110                 ds.getSubscriptionKey().addAll(keys);
111                 sub.deleteSubscription(ds);
112                 Assert.assertEquals(subscriptions.size(), keys.size());
113         }
