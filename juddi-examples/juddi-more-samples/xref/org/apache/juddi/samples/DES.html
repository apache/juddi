22  import java.security.spec.KeySpec;
23  import javax.crypto.BadPaddingException;
24  import javax.crypto.Cipher;
25  import javax.crypto.IllegalBlockSizeException;
26  import javax.crypto.KeyGenerator;
27  import javax.crypto.NoSuchPaddingException;
28  import javax.crypto.SecretKey;
29  import javax.crypto.SecretKeyFactory;
30  import javax.crypto.spec.DESedeKeySpec;
31  import org.apache.commons.codec.binary.Base64;
32  
33  /**
34   * This call is a simple test class that shows how to encrypt stuff in Java
35   * using 3DES
36   *
37   * @author <a href="mailto:alexoree@apache.org">Alex O'Ree</a>
38   * @deprecated  to be removed
39   */
40  @Deprecated
41  public class DES {
42  
43          private static final String UNICODE_FORMAT = "UTF8";
44          public static final String DESEDE_ENCRYPTION_SCHEME = "DESede";
45          static private KeySpec ks;
46          static private SecretKeyFactory skf;
47          static private Cipher cipher;
48          static byte[] arrayBytes;
49          static private String myEncryptionKey;
50          static private String myEncryptionScheme;
51          static SecretKey key;
52  
53          public DES() throws Exception {
54          }
55  
56          public String encrypt(String clear) throws NoSuchPaddingException, NoSuchAlgorithmException, InvalidAlgorithmParameterException, InvalidKeyException, IllegalBlockSizeException, BadPaddingException {
57                  String encryptedString = null;
58                  try {
59                          cipher.init(Cipher.ENCRYPT_MODE, key);
60                          byte[] plainText = clear.getBytes(UNICODE_FORMAT);
61                          byte[] encryptedText = cipher.doFinal(plainText);
62                          encryptedString = new String(Base64.encodeBase64(encryptedText), "UTF-8");
63                  } catch (Exception e) {
64                          e.printStackTrace();
65                  }
66                  return encryptedString;
67          }
68  
69          public String decrypt(String str) throws NoSuchPaddingException, NoSuchAlgorithmException, InvalidAlgorithmParameterException, InvalidKeyException, IllegalBlockSizeException, BadPaddingException {
70                  String encryptedString = null;
71                  try {
72                          cipher.init(Cipher.DECRYPT_MODE, key);
73                          byte[] encryptedText = Base64.decodeBase64(str.getBytes("UTF-8"));
74                          byte[] clear = cipher.doFinal(encryptedText);
75                          encryptedString = new String(Base64.encodeBase64(clear), "UTF-8");
76                  } catch (Exception e) {
77                          e.printStackTrace();
78                  }
79                  return encryptedString;
80          }
81  
82          public static void main(String[] args) throws Exception {
83                  DES des = new DES();
84                  KeyGenerator kgen;
85                  try {
86                          kgen = KeyGenerator.getInstance(DESEDE_ENCRYPTION_SCHEME);
87                          kgen.init(168);
88                          SecretKey skey = kgen.generateKey();
89                          byte[] raw = skey.getEncoded();
90                          myEncryptionKey = new String(Base64.encodeBase64(raw), "UTF-8");
91                          myEncryptionScheme = DESEDE_ENCRYPTION_SCHEME;
92                          arrayBytes = myEncryptionKey.getBytes(UNICODE_FORMAT);
93                          ks = new DESedeKeySpec(arrayBytes);
94                          skf = SecretKeyFactory.getInstance(myEncryptionScheme);
95                          cipher = Cipher.getInstance(myEncryptionScheme);
96                          key = skf.generateSecret(ks);
97  
98                          System.out.println(new String(Base64.encodeBase64(raw), "UTF-8"));
99                          System.out.println(des.encrypt("password"));
100                         System.out.println(des.decrypt(des.encrypt("password")));
101                 } catch (Exception ex) {
102                         ex.printStackTrace();;
103                 }
104 
105         }
106 }
