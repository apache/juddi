18  import javax.persistence.Column;
19  import javax.persistence.Entity;
20  import javax.persistence.FetchType;
21  import javax.persistence.GeneratedValue;
22  import javax.persistence.GenerationType;
23  import javax.persistence.Id;
24  import javax.persistence.JoinColumn;
25  import javax.persistence.ManyToOne;
26  import javax.persistence.Table;
27  
28  /**
29   * @author <a href="mailto:kurt@apache.org">Kurt T Stam</a>
30   * @author <a href="mailto:jfaath@apache.org">Jeff Faath</a>
31   */
32  @Entity
33  @Table(name = "j3_instance_details_doc_descr")
34  public class InstanceDetailsDocDescr implements java.io.Serializable {
35  
36  	private static final long serialVersionUID = -6936654678202381863L;
37  	private Long id;
38  	private TmodelInstanceInfo tmodelInstanceInfo;
39  	private String langCode;
40  	private String descr;
41  
42  	public InstanceDetailsDocDescr() {
43  	}
44  
45  	public InstanceDetailsDocDescr(Long id,
46  			TmodelInstanceInfo tmodelInstanceInfo, String descr) {
47  		this.id = id;
48  		this.tmodelInstanceInfo = tmodelInstanceInfo;
49  		this.descr = descr;
50  	}
51  
52  	public InstanceDetailsDocDescr(Long id,
53  			TmodelInstanceInfo tmodelInstanceInfo, String langCode, String descr) {
54  		this.id = id;
55  		this.tmodelInstanceInfo = tmodelInstanceInfo;
56  		this.langCode = langCode;
57  		this.descr = descr;
58  	}
59  
60  	@Id
61  	@GeneratedValue(strategy=GenerationType.AUTO)
62  	public Long getId() {
63  		return this.id;
64  	}
65  	public void setId(Long id) {
66  		this.id = id;
67  	}
68  
69  	@ManyToOne(fetch = FetchType.LAZY)
70  	@JoinColumn(name = "tmodel_instance_info_id", nullable = false)
71  	public TmodelInstanceInfo getTmodelInstanceInfo() {
72  		return this.tmodelInstanceInfo;
73  	}
74  	public void setTmodelInstanceInfo(TmodelInstanceInfo tmodelInstanceInfo) {
75  		this.tmodelInstanceInfo = tmodelInstanceInfo;
76  	}
77  
78  	@Column(name = "lang_code", length = 26)
79  	public String getLangCode() {
80  		return this.langCode;
81  	}
82  	public void setLangCode(String langCode) {
83  		this.langCode = langCode;
84  	}
85  
86  	@Column(name = "descr", nullable = false, length = 255)
87  	public String getDescr() {
88  		return this.descr;
89  	}
90  	public void setDescr(String descr) {
91  		this.descr = descr;
92  	}
93  
94  }
