47          private static UDDISecurityPortType security = null;
48          private static UDDIPublicationPortType publish = null;
49  
50          private static UDDIClient uddiClient = null;

52          /**
53           * This sets up the ws proxies using uddi.xml in META-INF
54           */
55          public SimpleCreateTmodelPartition() {
56                  try {
57                          // create a manager and read the config in the archive; 
58                          // you can use your config file name
59                          uddiClient = new UDDIClient("META-INF/partition-uddi.xml");
60                          uddiClient.start();
61  
62                          // a UddiClient can be a client to multiple UDDI nodes, so 
63                          // supply the nodeName (defined in your uddi.xml.
64                          // The transport can be WS, inVM, RMI etc which is defined in the uddi.xml
65                          Transport transport = uddiClient.getTransport("default");
66                          // Now you create a reference to the UDDI API
67  
68                          security = transport.getUDDISecurityService();
69                          publish = transport.getUDDIPublishService();
70                  } catch (Exception e) {
71                          e.printStackTrace();
72                  }
73          }

96          public void TmodelsTheEasyWay(String[] args) {
97                  try {
98                          
99                          //This reads from the config file
100                         UDDIClerk clerk = uddiClient.getClerk("defaultClerk");
101                         //Since the password isn't set in the above config file, we have to provide it manually
102                         //or thrown some fancy dialog box
103                         clerk.setPublisher("uddi");     //username
104                         clerk.setPassword("uddi");     //pass
105                         
106 
107                         TModel keygen = UDDIClerk.createKeyGenator("www.mycoolcompany.com", "My Company's Keymodel generator", "en");
108                         clerk.register(keygen);
109                         System.out.println("Creation of Partition Success!");
110 
111 
112                         //Now lets make a few tModels using the new domain
113                         TModel tm = new TModel();
114                         tm.setName(new Name());
115                         tm.getName().setValue("My Company's Department");
116                         tm.getName().setLang("en");
117                         tm.setTModelKey("uddi:www.mycoolcompany.com:department");
118                         clerk.register(tm);
119                         System.out.println("Creation of tModel Department Success!");
120 
121                         tm = new TModel();
122                         tm.setName(new Name());
123                         tm.getName().setValue("My Company's Authentication Method");
124                         tm.getName().setLang("en");
125                         tm.setTModelKey("uddi:www.mycoolcompany.com:authmode");
126                         clerk.register(tm);
127                         System.out.println("Creation of tModel Auth Mode Success!");
128 
129                         clerk.discardAuthToken();
130                 } catch (Exception e) {
131                         e.printStackTrace();
132                 }

135         public void TmodelsTheLongAndHardWay(String[] args) {
136                 try {
137                         //TODO this is where you, the developer need to find out if your UDDI server
138                         //supports UDDI's auth tokens or not. jUDDI does, so moving onwards.
139                         AuthStyle style = AuthStyle.UDDI_AUTH;
140                         //Login
141                         String key = GetAuthKey("uddi", "password", style);
142 
143                         //Note: when creating a tModel Key Generator, aka Partition, you MUST follow the below pattern
144                         //for jUDDI, the following is required
145                         //	Name
146                         //	CategoryBag/KR for the below fixed values
147                         //	a tModelKey that starts with uddi:<something>:keygenerator - all lower case
148                         //First, Here's the long way to do it to make a Key Generator tModel.
149                         SaveTModel st = new SaveTModel();
150                         st.setAuthInfo(key);
151                         TModel tm = new TModel();
152                         tm.setName(new Name());
153                         tm.getName().setValue("My Company's Keymodel generator");
154                         tm.getName().setLang("en");
155                         tm.setCategoryBag(new CategoryBag());
156                         KeyedReference kr = new KeyedReference();
157                         kr.setTModelKey("uddi:uddi.org:categorization:types");
158                         kr.setKeyName("uddi-org:keyGenerator");
159                         kr.setKeyValue("keyGenerator");
160                         tm.getCategoryBag().getKeyedReference().add(kr);
161                         tm.setTModelKey("uddi:www.mycoolcompany.com:keygenerator");
162                         st.getTModel().add(tm);
163                         TModelDetail saveTModel = publish.saveTModel(st);
164                         System.out.println("Creation of Partition Success!");
165

199         private enum AuthStyle {
200 
201                 HTTP,
202                 UDDI_AUTH
203 
204         }
205 
206         /**
207          * Gets a UDDI style auth token, otherwise, appends credentials to the
208          * ws proxies (not yet implemented)
209          *
210          * @param username
211          * @param password
212          * @param style
213          * @return
214          */
215         private String GetAuthKey(String username, String password, AuthStyle style) {
216                 switch (style) {
217                         case HTTP:
218                                 ((BindingProvider) publish).getRequestContext().put(BindingProvider.USERNAME_PROPERTY, username);
219                                 ((BindingProvider) publish).getRequestContext().put(BindingProvider.PASSWORD_PROPERTY, password);
220                                 return null;
221                         case UDDI_AUTH:
222                                 try {
223 
224                                         GetAuthToken getAuthTokenRoot = new GetAuthToken();
225                                         getAuthTokenRoot.setUserID(username);
226                                         getAuthTokenRoot.setCred(password);
227 
228                                         // Making API call that retrieves the authentication token for the 'root' user.
229                                         AuthToken rootAuthToken = security.getAuthToken(getAuthTokenRoot);
230                                         System.out.println(username + " AUTHTOKEN = (don't log auth tokens!)");
231                                         return rootAuthToken.getAuthInfo();
232                                 } catch (Exception ex) {
233                                         System.out.println("Could not authenticate with the provided credentials " + ex.getMessage());
234                                 }
235                 }
236 
237                 return null;
238         }
