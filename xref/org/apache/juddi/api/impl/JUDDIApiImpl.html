144         /**
145          * Saves publisher(s) to the persistence layer. This method is specific
146          * to jUDDI. Administrative privilege required.
147          *
148          * @param body
149          * @return PublisherDetail
150          * @throws DispositionReportFaultMessage
151          */
152         public PublisherDetail savePublisher(SavePublisher body)
153                 throws DispositionReportFaultMessage {
154                 long startTime = System.currentTimeMillis();
155                 EntityManager em = PersistenceManager.getEntityManager();
156                 EntityTransaction tx = em.getTransaction();
157                 try {
158                         tx.begin();
159 
160                         UddiEntityPublisher publisher = this.getEntityPublisher(em, body.getAuthInfo());
161 
162                         new ValidatePublish(publisher).validateSavePublisher(em, body);
163 
164                         PublisherDetail result = new PublisherDetail();
165 
166                         List<org.apache.juddi.api_v3.Publisher> apiPublisherList = body.getPublisher();
167                         for (org.apache.juddi.api_v3.Publisher apiPublisher : apiPublisherList) {
168 
169                                 org.apache.juddi.model.Publisher modelPublisher = new org.apache.juddi.model.Publisher();
170 
171                                 MappingApiToModel.mapPublisher(apiPublisher, modelPublisher);
172 
173                                 Object existingUddiEntity = em.find(modelPublisher.getClass(), modelPublisher.getAuthorizedName());
174                                 if (existingUddiEntity != null) {
175                                         em.remove(existingUddiEntity);
176                                 }
177 
178                                 em.persist(modelPublisher);
179 
180                                 result.getPublisher().add(apiPublisher);
181                         }
182 
183                         tx.commit();
184                         long procTime = System.currentTimeMillis() - startTime;
185                         serviceCounter.update(JUDDIQuery.SAVE_PUBLISHER,
186                                 QueryStatus.SUCCESS, procTime);
187                         return result;
188                 } catch (DispositionReportFaultMessage drfm) {
189                         long procTime = System.currentTimeMillis() - startTime;
190                         serviceCounter.update(JUDDIQuery.SAVE_PUBLISHER,
191                                 QueryStatus.FAILED, procTime);
192                         throw drfm;
193                 } finally {
194                         if (tx.isActive()) {
195                                 tx.rollback();
196                         }
197                         em.close();
198                 }

201         /**
202          * Deletes publisher(s) from the persistence layer. This method is
203          * specific to jUDDI. Administrative privilege required. Also removes
204          * all registered business entities of the user and marks all created
205          * tModels as "deleted" but not does not remove the tModel from
206          * persistence. All subscriptions are also destroyed
207          *
208          * @param body
209          * @throws DispositionReportFaultMessage
210          */
211         @Override
212         public void deletePublisher(DeletePublisher body)
213                 throws DispositionReportFaultMessage {
214                 long startTime = System.currentTimeMillis();
215                 EntityManager em = PersistenceManager.getEntityManager();
216                 EntityTransaction tx = em.getTransaction();
217                 try {
218                         tx.begin();
219 
220                         UddiEntityPublisher publisher = this.getEntityPublisher(em, body.getAuthInfo());
221 
222                         new ValidatePublish(publisher).validateDeletePublisher(em, body);
223 
224                         List<String> entityKeyList = body.getPublisherId();
225                         List<Publisher> deletedPubs = new ArrayList<Publisher>();
226                         for (String entityKey : entityKeyList) {
227                                 Publisher obj = em.find(org.apache.juddi.model.Publisher.class, entityKey);
228                                 deletedPubs.add(obj);
229                                 //get an authtoken for this publisher so that we can get its registeredInfo
230                                 UDDISecurityImpl security = new UDDISecurityImpl();
231                                 AuthToken authToken = security.getAuthToken(entityKey);
232 
233                                 GetRegisteredInfo r = new GetRegisteredInfo();
234                                 r.setAuthInfo(authToken.getAuthInfo());
235                                 r.setInfoSelection(InfoSelection.ALL);
236 
237                                 log.info("removing all businesses owned by publisher " + entityKey + ".");
238                                 UDDIPublicationImpl publish = new UDDIPublicationImpl();
239                                 RegisteredInfo registeredInfo = publish.getRegisteredInfo(r);
240                                 BusinessInfos businessInfos = registeredInfo.getBusinessInfos();
241                                 if (businessInfos != null && businessInfos.getBusinessInfo() != null) {
242                                         Iterator<BusinessInfo> iter = businessInfos.getBusinessInfo().iterator();
243                                         while (iter.hasNext()) {
244                                                 BusinessInfo businessInfo = iter.next();
245                                                 Object business = em.find(org.apache.juddi.model.BusinessEntity.class, businessInfo.getBusinessKey());
246                                                 em.remove(business);
247                                         }
248                                 }
249 
250                                 log.info("mark all tmodels for publisher " + entityKey + " as deleted.");
251                                 TModelInfos tmodelInfos = registeredInfo.getTModelInfos();
252                                 if (tmodelInfos != null && tmodelInfos.getTModelInfo() != null) {
253                                         Iterator<TModelInfo> iter = tmodelInfos.getTModelInfo().iterator();
254                                         while (iter.hasNext()) {
255                                                 TModelInfo tModelInfo = iter.next();
256                                                 Tmodel tmodel = (Tmodel) em.find(org.apache.juddi.model.Tmodel.class, tModelInfo.getTModelKey());
257                                                 tmodel.setDeleted(true);
258                                                 em.persist(tmodel);
259                                         }
260                                 }
261                                 log.info("remove all persisted AuthTokens for publisher " + entityKey + ".");
262                                 Query q1 = em.createQuery("DELETE FROM AuthToken auth WHERE auth.authorizedName = ?1");
263                                 q1.setParameter(1, entityKey);
264                                 q1.executeUpdate();
265                                 log.info("remove all subscriptions for publisher " + entityKey + ".");
266                                 q1 = em.createQuery("DELETE FROM Subscription s WHERE s.authorizedName = ?1");
267                                 q1.setParameter(1, entityKey);
268                                 q1.executeUpdate();
269 
270                                 log.info("removing publisher " + entityKey + ".");
271                                 //delete the publisher
272                                 em.remove(obj);
273                         }
274 
275                         tx.commit();
276                         for (Publisher p: deletedPubs){
277                                 USERFRIENDLYSMTPNotifier.notifyAccountDeleted(new TemporaryMailContainer(null, p, (Publisher) publisher));
278                         }
279                         long procTime = System.currentTimeMillis() - startTime;
280                         serviceCounter.update(JUDDIQuery.DELETE_PUBLISHER,
281                                 QueryStatus.SUCCESS, procTime);
282                 } catch (DispositionReportFaultMessage drfm) {
283                         long procTime = System.currentTimeMillis() - startTime;
284                         serviceCounter.update(JUDDIQuery.DELETE_PUBLISHER,
285                                 QueryStatus.FAILED, procTime);
286                         throw drfm;
287                 } finally {
288                         if (tx.isActive()) {
289                                 tx.rollback();
290                         }
291                         em.close();
292                 }
293         }
